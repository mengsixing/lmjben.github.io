function throttle(callback, timeout) {
  var first = true;
  var throttleId;
  return function(...rest) {
    if (first) {
      callback.apply(this, rest);
      throttleId = null;
      first = false;
    } else if (!throttleId) {
      throttleId = setTimeout(() => {
        callback.apply(this, rest);
        throttleId = null;
      }, timeout);
    }
  };
}
