function insertSort() {
  var temp, inner;
  for (var outer = 1; outer < this.dataSource.length; outer++) {
    temp = this.dataSource[outer];
    inner = outer;
    while (inner > 0 && this.dataSource[inner - 1] >= temp) {
      this.dataSource[inner] = this.dataSource[inner - 1];
      inner--;
    }
    this.dataSource[inner] = temp;

}
